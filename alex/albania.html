Equality for all: Only in Albania

student.sub = student2012 %>% 
  select(CNT, OUTHOURS, math) %>% filter(!is.na(OUTHOURS))

qplot(x = OUTHOURS, y = math, 
      data = student.sub, facets = ~CNT, 
      alpha = I(0.05))

qplot(x = OUTHOURS, y = math, 
      data = student.sub %>% filter(CNT == "Albania"), 
      alpha = I(0.05))
